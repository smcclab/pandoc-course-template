% ── Manual proof-of-concept for the BeforeBeginEnvironment + flag approach ───
% This file mirrors what beamer-background.lua injects into the preamble and
% document body.  Compile with lualatex to verify visually.
%
% Expected results:
%   Slide 1  (title)     — default theme background, NO image
%   Slide 2              — default theme background, NO image
%   Slide 3              — sullivans.jpg full-cover, opaque
%   Slide 4              — default theme background restored
%   Slide 5              — sullivans.jpg at 35% opacity (semi-transparent)
%   Slide 6              — default theme background restored
%   Slide 7              — sullivans.jpg contain (letterboxed)
%   Slide 8              — default theme background restored

\usepackage{etoolbox}
\usepackage{tikz}

\newif\ifbgpending  % true → next \begin{frame} should receive a background image
\newif\ifbgopaque   % true → opaque; false → semi-transparent TikZ node
\newif\ifbgactive   % true → a background is applied; reset on next plain frame
\global\bgpendingfalse
\global\bgopaquetrue
\global\bgactivefalse
\gdef\bgpendingimage{}
\gdef\bgpendingsize{cover}
\gdef\bgpendingopc{1}

\BeforeBeginEnvironment{frame}{%
  \ifbgpending
    \def\tmpbgsize{\bgpendingsize}%
    \def\tmpcontain{contain}%
    \ifbgopaque
      \ifx\tmpbgsize\tmpcontain
        \setbeamertemplate{background}{%
          \begin{tikzpicture}[remember picture,overlay]
            \node at (current page.center)
              {\includegraphics[width=\paperwidth,height=\paperheight,
                                keepaspectratio]{\bgpendingimage}};
          \end{tikzpicture}}%
      \else
        \setbeamertemplate{background}{%
          \begin{tikzpicture}[remember picture,overlay]
            \node at (current page.center)
              {\includegraphics[width=\paperwidth,height=\paperheight]%
                {\bgpendingimage}};
          \end{tikzpicture}}%
      \fi
    \else
      \ifx\tmpbgsize\tmpcontain
        \setbeamertemplate{background}{%
          \begin{tikzpicture}[remember picture,overlay]
            \node[opacity=\bgpendingopc] at (current page.center)
              {\includegraphics[width=\paperwidth,height=\paperheight,
                                keepaspectratio]{\bgpendingimage}};
          \end{tikzpicture}}%
      \else
        \setbeamertemplate{background}{%
          \begin{tikzpicture}[remember picture,overlay]
            \node[opacity=\bgpendingopc] at (current page.center)
              {\includegraphics[width=\paperwidth,height=\paperheight]%
                {\bgpendingimage}};
          \end{tikzpicture}}%
      \fi
    \fi
    \global\bgpendingfalse
    \global\bgopaquetrue
    \global\bgactivetrue
  \else
    \ifbgactive
      \setbeamertemplate{background}{}%
      \global\bgactivefalse
    \fi
  \fi
}
